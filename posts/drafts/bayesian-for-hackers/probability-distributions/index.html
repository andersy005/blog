colors = ["#348ABD", "#A60628"]

def poisson_dist(位=1.5):
    fig, ax = plt.subplots(aspect=(3.5, 2), width=6)
    ax.bar(
        a,
        scipy.stats.poisson.pmf(a, 位),
        color=colors[0],
        alpha=0.60,
        edgecolor=colors[0],
        lw="3",
        label=fr"$\lambda ={位}",
    )
    ax.format(
        xticks=a,
        ylabel="Probability of $k$",
        xlabel="$k$",
        title="Probability mass function of a Poisson random variable",
    )
    plt.close(fig)
    return fig
